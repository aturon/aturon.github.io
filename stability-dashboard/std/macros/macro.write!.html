std::macros::write!Experimental[src]

macro_rules! write(
    ($dst:expr, $($arg:tt)*) => ({
        format_args_method!($dst, write_fmt, $($arg)*)
    })
)

Use the format! syntax to write data into a buffer of type &mut Writer. See std::fmt for more information.

Example

fn main() { #![allow(unused_must_use)] use std::io::MemWriter; let mut w = MemWriter::new(); write!(&mut w, "test"); write!(&mut w, "formatted {}", "arguments"); }
use std::io::MemWriter;

let mut w = MemWriter::new();
write!(&mut w, "test");
write!(&mut w, "formatted {}", "arguments");
