The normal distribution N(mean, std_dev**2).

This uses the ZIGNOR variant of the Ziggurat method, see StandardNormal for more details.

Example

fn main() { use std::rand; use std::rand::distributions::{Normal, IndependentSample}; // mean 2, standard deviation 3 let normal = Normal::new(2.0, 3.0); let v = normal.ind_sample(&mut rand::task_rng()); println!("{} is from a N(2, 9) distribution", v) }
use std::rand;
use std::rand::distributions::{Normal, IndependentSample};

// mean 2, standard deviation 3
let normal = Normal::new(2.0, 3.0);
let v = normal.ind_sample(&mut rand::task_rng());
println!("{} is from a N(2, 9) distribution", v)
