The exponential distribution Exp(lambda).

This distribution has density function: f(x) = lambda * exp(-lambda * x) for x > 0.

Example

fn main() { use std::rand; use std::rand::distributions::{Exp, IndependentSample}; let exp = Exp::new(2.0); let v = exp.ind_sample(&mut rand::task_rng()); println!("{} is from a Exp(2) distribution", v); }
use std::rand;
use std::rand::distributions::{Exp, IndependentSample};
