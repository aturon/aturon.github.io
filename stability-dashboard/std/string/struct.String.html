fn from_utf8(vec: Vec<u8>) -> Result<String, Vec<u8>>

Returns the vector as a string buffer, if possible, taking care not to copy it.

Returns Err with the original vector if the vector contains invalid UTF-8.

Example

fn main() { let hello_vec = vec![104, 101, 108, 108, 111]; let string = String::from_utf8(hello_vec); assert_eq!(string, Ok("hello".to_string())); }
let hello_vec = vec![104, 101, 108, 108, 111];
let string = String::from_utf8(hello_vec);
assert_eq!(string, Ok("hello".to_string()));

fn from_utf8_lossy(v: &'a [u8]) -> MaybeOwned<'a>

Converts a vector of bytes to a new utf-8 string. Any invalid utf-8 sequences are replaced with U+FFFD REPLACEMENT CHARACTER.

Example

fn main() { let input = b"Hello \xF0\x90\x80World"; let output = String::from_utf8_lossy(input); assert_eq!(output.as_slice(), "Hello \uFFFDWorld"); }
let input = b"Hello \xF0\x90\x80World";
let output = String::from_utf8_lossy(input);
assert_eq!(output.as_slice(), "Hello \uFFFDWorld");

fn from_utf16_lossy(v: &[u16]) -> String
