Concatenate a vector of strings.

fn connect(&self, sep: &str) -> String

Concatenate a vector of strings, placing a given separator between each.

Example

fn main() { let first = "Roast".to_string(); let second = "Sirloin Steak".to_string(); let string_vec = vec![first, second]; assert_eq!(string_vec.connect(", "), "Roast, Sirloin Steak".to_string()); }
let first = "Roast".to_string();
let second = "Sirloin Steak".to_string();
let string_vec = vec![first, second];
assert_eq!(string_vec.connect(", "), "Roast, Sirloin Steak".to_string());
